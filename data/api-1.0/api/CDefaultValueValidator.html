CDefaultValueValidator sets the attributes with the specified value. It does not do validation. Its existence is mainly to allow specifying attribute default values in a dynamic way.

protected function validateAttribute($object,$attribute)
{
    if(!$this->setOnEmpty)
        $object->$attribute=$this->value;
    else
    {
        $value=$object->$attribute;
        if($value===null || $value==='')
            $object->$attribute=$this->value;
    }
}
